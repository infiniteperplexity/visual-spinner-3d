Using the VisualSpinner Engine, someone with basic programming knowledge can easily create complex patterns.

Here is the code that drives the entire pattern:

for (var i = 0; i<8; i++) {

    propOne.addMove(moves.build.flower({direction: CLOCKWISE, entry: propOne.socket().hand.angle, spin: vs.ANTISPIN, duration: (i==4) ? 1 : 1.25}));

    propTwo.addMove(moves.build.flower({direction: 1-(i%2)*2, entry: propTwo.socket().hand.angle, spin: vs.ANTISPIN, duration: (i==4) ? 1 : 1.25}));

    propThree.addMove(moves.build.flower({direction: COUNTERCLOCKWISE, entry: propThree.socket().hand.angle, spin: vs.ANTISPIN, duration: (i==4) ? 1 : 1.25}));

    propFour.addMove(moves.build.flower({direction: (i%2)*2-1, entry: propFour.socket().hand.angle, spin: vs.ANTISPIN, duration: (i==4) ? 1 : 1.25}));

}
